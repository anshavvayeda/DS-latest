# =============================================================================
# EC2 + RDS DEPLOYMENT - EXACT COMMANDS
# =============================================================================
# Run these commands in order on your EC2 instance
# =============================================================================

#####################################################
# PHASE 1: EC2 INITIAL SETUP (Run once on new EC2)
#####################################################

# 1. SSH into your EC2
ssh -i your-key.pem ubuntu@YOUR_EC2_PUBLIC_IP

# 2. Update system
sudo apt update && sudo apt upgrade -y

# 3. Install dependencies
sudo apt install -y python3 python3-pip python3-venv python3-dev git nginx curl htop build-essential libpq-dev postgresql-client

# 4. Create app directory
sudo mkdir -p /home/ubuntu/studybuddy
sudo chown -R ubuntu:ubuntu /home/ubuntu/studybuddy

# 5. Clone your repository (or upload files)
cd /home/ubuntu/studybuddy
git clone YOUR_REPO_URL .
# OR: Use scp to upload files
# scp -i your-key.pem -r ./backend ubuntu@YOUR_EC2_IP:/home/ubuntu/studybuddy/

# 6. Create Python virtual environment
cd /home/ubuntu/studybuddy/backend
python3 -m venv venv
source venv/bin/activate

# 7. Install Python dependencies
pip install --upgrade pip
pip install -r requirements.txt

# 8. Create required directories
mkdir -p logs uploads

#####################################################
# PHASE 2: CONFIGURE ENVIRONMENT
#####################################################

# 9. Create .env file (REQUIRED - fill in your values)
cat > /home/ubuntu/studybuddy/backend/.env << 'EOF'
# Database
DATABASE_URL=postgresql+asyncpg://postgres:abv84012@database-1.c7iya4u0apvr.ap-south-1.rds.amazonaws.com:5432/studybuddy

# Authentication
JWT_SECRET=7c96ec43e36b1ba78f0e6d129e09591d68f80c039efb3b5ac3fe639086407082
JWT_ALGORITHM=HS256
JWT_EXPIRATION_HOURS=24

# Admin
ADMIN_USERNAME=admin
ADMIN_PASSWORD=YOUR_ADMIN_PASSWORD

# AWS S3
AWS_ACCESS_KEY_ID=YOUR_AWS_KEY
AWS_SECRET_ACCESS_KEY=YOUR_AWS_SECRET
S3_BUCKET_NAME=YOUR_BUCKET
AWS_REGION=ap-south-1

# AI Service
OPENROUTER_API_KEY=YOUR_OPENROUTER_KEY

# Production Settings
ENV=production
MOCK_OTP_MODE=false
EOF

# 10. Secure the .env file
chmod 600 /home/ubuntu/studybuddy/backend/.env

#####################################################
# PHASE 3: DATABASE SETUP
#####################################################

# 11. Test database connection
cd /home/ubuntu/studybuddy/backend
source venv/bin/activate
python3 -c "
import asyncio
from sqlalchemy.ext.asyncio import create_async_engine
from sqlalchemy import text
import os
from dotenv import load_dotenv
load_dotenv()

async def test():
    engine = create_async_engine(os.getenv('DATABASE_URL'))
    async with engine.connect() as conn:
        result = await conn.execute(text('SELECT 1'))
        print('✅ Database connection successful!')
    await engine.dispose()

asyncio.run(test())
"

# 12. Run Alembic migrations
alembic upgrade head

# 13. Verify migrations
alembic current

# 14. (Optional) Seed initial data
python init_data.py

#####################################################
# PHASE 4: SETUP SYSTEMD SERVICE
#####################################################

# 15. Copy service file
sudo cp /home/ubuntu/studybuddy/backend/deploy/studybuddy.service /etc/systemd/system/

# 16. Reload systemd
sudo systemctl daemon-reload

# 17. Enable service (start on boot)
sudo systemctl enable studybuddy

# 18. Start service
sudo systemctl start studybuddy

# 19. Check status
sudo systemctl status studybuddy

# 20. View logs
sudo journalctl -u studybuddy -f

#####################################################
# PHASE 5: SETUP NGINX
#####################################################

# 21. Copy nginx config (update domain in file first if needed)
sudo cp /home/ubuntu/studybuddy/backend/deploy/nginx.conf /etc/nginx/sites-available/studybuddy

# 22. Enable site
sudo ln -sf /etc/nginx/sites-available/studybuddy /etc/nginx/sites-enabled/

# 23. Remove default site
sudo rm -f /etc/nginx/sites-enabled/default

# 24. Test nginx config
sudo nginx -t

# 25. Restart nginx
sudo systemctl restart nginx

#####################################################
# PHASE 6: VERIFY DEPLOYMENT
#####################################################

# 26. Test health endpoint (local)
curl http://localhost:8000/health

# 27. Test health endpoint (via nginx)
curl http://localhost/health

# 28. Test from outside (use EC2 public IP)
curl http://YOUR_EC2_PUBLIC_IP/health

#####################################################
# PHASE 7: (OPTIONAL) SSL WITH LET'S ENCRYPT
#####################################################

# 29. Install certbot
sudo apt install certbot python3-certbot-nginx

# 30. Get certificate (replace with your domain)
sudo certbot --nginx -d your-domain.com

# 31. Test auto-renewal
sudo certbot renew --dry-run

#####################################################
# MAINTENANCE COMMANDS
#####################################################

# Restart service
sudo systemctl restart studybuddy

# View logs
sudo journalctl -u studybuddy -n 100

# Follow logs
sudo journalctl -u studybuddy -f

# Deploy updates
cd /home/ubuntu/studybuddy/backend
./deploy/deploy.sh

# Quick restart only
./deploy/deploy.sh --restart-only

psql -h database-1.c7iya4u0apvr.ap-south-1.rds.amazonaws.com -U postgres -d studybuddy -c "SELECT count(*) FROM pg_stat_activity;"

df -h

# Check memory
